    代码示例 
    <template>
  <div style="margin-top: 16px">
    <TmxTransferRight
        selectKey="title"
        :selectedData="initData"
        transferHeight="300px"
        :drag="true"
        @on-remove-all="removeAllHandler"
        @on-drag-update="dragUpdateHandler"
        @on-remove-item="removeItemHandler"
        @on-change="onSelectChange"
        @on-item-click="onItemClick"
    >
    </TmxTransferRight>
  </div>
</template>

    export default {
  data() {
    return {
      tableHeight: 250,
      initData: [
        { title: "Data Name3", model: "TMX06", disabled: true },
        {
          title: "Data Name4",
          model: "TMX05",
          description: "desciption",
        },
        {
          title: "Data Name7",
          model: "TMX05",
        },
      ],
    };
  },
  methods: {
    removeAllHandler() {
      console.log("清空数据");
    },
    dragUpdateHandler(data1, data2) {
      console.log("拖拽数据", data1, data2);
    },
    removeItemHandler(item, index) {
      console.log("删除单个数据", item, index);
    },
    onSelectChange(data) {
      console.log("右侧数据发生改变", data);
    },
    onItemClick(item) {
      console.log("点击了某一列", item);
    },
  },
};
