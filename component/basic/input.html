    # 输入框 Input

    # 基础用法

    <Input  placeholder="Basic" :clearable="true" />
<Input  placeholder="Basic" :clearable="true" value="Basic Text" />
<Input  placeholder="Disabled" :clearable="true" disabled/>
<Input  placeholder="Disabled" :clearable="true" value="Basic Text" disabled />
<Input  prefix="Search" placeholder="Basic" :clearable="true" />
<Input  suffix="Search" placeholder="Basic" :clearable="true" />

    # URL 地址输入框

    Http://
    .com
    <Input  placeholder="Enter URL" style="width: 400px">
    <span slot="prepend">Http://</span>
    <span slot="append">.com</span>
</Input>

    # 带后缀单位的输入框1 （此方式有局限性，最多显示六个字符，设置清空，会将suffix的dom删除，因此设置clearable将无法同时查看单位，额外需添加一个 tmx-unit class）

    km/h 
    <Input class="tmx-input tmx-unit" placeholder="Enter" style="width: 400px">
    <span slot="suffix">km/h</span>
</Input>

    # 带后缀单位的输入框 2

    Hour
    <Input  placeholder="Enter" style="width: 400px">
    <span slot="append">Hour</span>
</Input>

    # 带前缀单位的输入框

    Select
    • No Data
    • 86
      Loading
    <Input  placeholder="Enter" :clearable="true" style="width: 400px">
    <span slot="prepend">
      <Select value="86" style="width: 60px;">
        <Option value="86">86</Option>
      </Select>
    </span>
</Input>
