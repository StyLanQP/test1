    # 复选框 CheckBox

    # 基础用法

    <Checkbox disabled :value="true">电风扇1</Checkbox>
<Checkbox disabled :value="false">电风扇2<Checkbox>
<Checkbox :value="true">电风扇3</Checkbox>
<Checkbox :value="false">电风扇4</Checkbox>
<CheckboxGroup v-model="fruit">
    <Checkbox label="香蕉"></Checkbox>
    <Checkbox label="苹果"></Checkbox>
    <Checkbox label="西瓜"></Checkbox>
</CheckboxGroup>

    # 文案在左,选择在右

    <div style="margin-top: 16px">
    <Checkbox typeValue="right" disabled :value="true">电风扇1</Checkbox>
    <Checkbox typeValue="right" disabled :value="false">电风扇2</Checkbox>
    <Checkbox typeValue="right" :value="true">电风扇3</Checkbox>
    <Checkbox typeValue="right" :value="false">电风扇4</Checkbox>
    <CheckboxGroup v-model="fruit">
        <Checkbox typeValue="right" label="香蕉"></Checkbox>
        <Checkbox typeValue="right" label="苹果"></Checkbox>
        <Checkbox typeValue="right" label="西瓜"></Checkbox>
        <Checkbox typeValue="right" label="榴莲" disabled></Checkbox>
        <Checkbox typeValue="right" label="莲雾" disabled></Checkbox>
    </CheckboxGroup>
</div>
