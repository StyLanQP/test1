    # 说明

    对指定列设置 searchable: true，开启表头搜索功能。
    表头搜索有三种模式：基础搜索、固定类别搜索与无固定类别搜索。

    # 基础搜索

    单选模式，需要设置 searchMultiple: false
    获取远程数据，异步方法传入 searchDataSource
    选中某一项数据后触发 searchRemote

    {
  title: 'Name',
  key: 'name',
  searchable: true,
  searchMultiple: false,
  searchDataSource (val) {
    return new Promise(resolve => {
      const data = [
        { value: 'John Brown', label: 'John Brown' },
        { value: 'Jim Green', label: 'Jim Green' },
        { value: 'Joe Black', label: 'Joe Black' },
        { value: 'Jon Snow', label: 'Jon Snow' },
      ]
      const res = data.filter(item => item.label.indexOf(val.toLowerCase()) > -1)
      resolve(res)
    })
  },
  searchRemote(value, row){
    console.log('data callback')
  }
},

    # 固定类别搜索

    固定类别数组传入 searchData
    选中某一项数据后触发 searchRemote

    {
  title: 'Age',
  key: 'age',
  searchable: true,
  searchData:[
    { value: '18', label: '18' },
    { value: '24', label: '24' },
    { value: '30', label: '30' },
    { value: '26', label: '26' }
  ],
  searchRemote(value, row){
    console.log('callback', value, row)
  }
},

    # 无固定类别搜索

    获取远程数据，异步方法传入 searchDataSource 选中某一项数据后触发 searchRemote

    {
  title: 'Address',
  key: 'address',
  searchable: true,
  searchDataSource (val) {
    return new Promise(resolve => {
      const data = [

        { value: 'London No. 1 Lake Park', label: 'London No. 1 Lake Park' },
        { value: 'Sydney No. 1 Lake Park', label: 'Sydney No. 1 Lake Park' },
        { value: 'Ottawa No. 2 Lake Park', label: 'Ottawa No. 2 Lake Park' }
      ]
      const res = data.filter(item => item.label.indexOf(val.toLowerCase()) > -1)
      resolve(res)
    })
  },
  searchRemote(value, row){
    console.log('callback', value, row)
  }
}
