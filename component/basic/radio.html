    # 单选框 Radio

    # 基础用法

    <RadioGroup value="爪哇犀牛">
  <Radio class="tmx-raido" label="金斑蝶"></Radio>
  <Radio class="tmx-raido" label="爪哇犀牛"></Radio>
  <Radio class="tmx-raido" label="印度黑羚" disabled></Radio>
  <Radio class="tmx-raido" label="爪哇犀牛" disabled></Radio>
</RadioGroup>

    # 带边框的单选

    <RadioGroup value="爪哇犀牛">
  <Radio class="tmx-raido" label="金斑蝶"></Radio>
  <Radio class="tmx-raido" label="爪哇犀牛"></Radio>
  <Radio class="tmx-raido" label="印度黑羚" disabled></Radio>
  <Radio class="tmx-raido" label="爪哇犀牛" disabled></Radio>
</RadioGroup>

    # 按钮样式

    <RadioGroup class="tmx-radio-group"  v-model="button1" type="button">
    <Radio label="北京"></Radio>
    <Radio label="上海"></Radio>
    <Radio label="深圳"></Radio>
    <Radio label="杭州"></Radio>
</RadioGroup>
