    # 单选框 Radio

    # 基础用法

    <RadioGroup value="爪哇犀牛">
    <Radio label="金斑蝶"></Radio>
    <Radio label="爪哇犀牛"></Radio>
    <Radio label="Option 1"></Radio>
    <Radio label="Option 2"></Radio>
    <Radio label="印度黑羚" disabled></Radio>
    <Radio label="爪哇犀牛" disabled></Radio>
    <Radio label="Option 3" disabled></Radio>
    <Radio label="Option 4" disabled></Radio>
</RadioGroup>

    # 按钮样式

    <RadioGroup v-model="button1" type="button">
    <Radio label="北京"></Radio>
    <Radio label="上海"></Radio>
    <Radio label="深圳"></Radio>
    <Radio label="杭州"></Radio>
</RadioGroup>

    # 用于容器Header的按钮样式

    <RadioGroup v-model="button1" type="header">
    <Radio label="北京"></Radio>
    <Radio label="上海"></Radio>
    <Radio label="深圳"></Radio>
    <Radio label="杭州"></Radio>
</RadioGroup>
