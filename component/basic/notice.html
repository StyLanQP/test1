    # 基础使用

    代码示例 
    <template>
  <div>
    <button @click="info(false)">Info</button>
    <button @click="success(false)">Success</button>
    <button @click="warning(false)">Warning</button>
    <button @click="error(false)">Error</button>
  </div>
</template>
<script>
    export default {
        methods: {
          info(nodesc) {
            this.$Notice.info({
              title: "Info",
              duration: 15,
              desc: nodesc ? "" : "Here is the notification. ",
              onClose: this.onClose,
            });
          },
          success(nodesc) {
            this.$Notice.success({
              title: "Notification title",
              duration: 0,
              desc: nodesc
                ? ""
                : "Here is the notification description. Here is the notification description. ",
            });
          },
          warning(nodesc) {
            this.$Notice.warning({
              title: "Notification title",
              duration: 0,
              desc: nodesc
                ? ""
                : "Here is the notification description. Here is the notification description. ",
            });
          },
          error(nodesc) {
            this.$Notice.error({
              title: "Notification title",
              duration: 0,
              desc: nodesc
                ? ""
                : "Here is the notification description. Here is the notification description. ",
            });
          },
        },
    }
</script>

    # 自定义页头和页脚

    代码示例 
    <template>
  <div>
    <Button @click="info1(false)">带确定按钮的页脚</Button>
    <Button @click="info2(false)">自定义页脚</Button>
  </div>
</template>
<script>
import SlotFooter from "./slotFooter.vue";

export default {
  methonds: {
    info1(nodesc) {
      this.$Notice.info({
        title: "Info",
        duration: 15,
        showConfirm: true,
        desc: nodesc ? "" : "Here is the notification. ",
        onClose: this.onClose,
      });
    }, 
    info2(nodesc) {
      this.$Notice.info({
        title: "Info",
        duration: 15,
        footerRender: (h) => {
          return h(SlotFooter);
        },
        desc: nodesc ? "" : "Here is the notification. ",
      });
    }, 
    onClose(val) {
      console.log("bbbbbb", val);
    },   
  }
}
</script>

    # 页头显示Icon

    代码示例 
    <template>
  <div>
    <Button @click="success1(false)">带有icon的页头</Button>
    <Button @click="success2(false)">自定义icon</Button>
  </div>
</template>
<script>
export default {
  methonds: {
    sucess1(nodesc) {
      this.$Notice.success({
        title: "Notification title",
        duration: 0,
        showTitleIcon: true,
        desc: nodesc
          ? ""
          : "Here is the notification description. Here is the notification description. ",
      });
    }, 
    sucess2(nodesc) {
      this.$Notice.success({
        title: "Notification title",
        duration: 0,
        showTitleIcon: true,
        titleIcon: "AlarmAction",
        desc: nodesc
          ? ""
          : "Here is the notification description. Here is the notification description. ",
      });
    }, 
    onClose(val) {
      console.log("bbbbbb", val);
    },   
  }
}
</script>

    # Notice props

    属性
    属性
    说明
    说明
    类型
    类型
    默认值
    默认值
    showConfirm
    是否显示页脚的确定按钮, 勾选的值会在onClose回调函数返回
    Boolean
    false
    showTitleIcon
    是否显示title的icon（目前还不支持title自定义，后续有需求再看）
    Boolean
    false
    titleIcon
    title左边icon自定义，和Icon组件使用一样（只有在为true的情况才会生效）
    String
    -
    footerRender
    自定义描述页脚，使用 Vue 的 Render 函数，如果同时设置了 footerRender 和 showConfirm，则只显示 footerRender 的内容
    Function
