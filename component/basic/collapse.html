      <div class="title" style="display: flex;">
        <div style="flex:1">
          <p style="line-height: 21px;margin-bottom: 4px;">

            Child of the first accordion. Child of the first accordion.

          Save
        </button>
      </div>
      <p slot="content">内容</p>
    </Panel>
    <Panel name="5">
      嵌套折叠
      <Collapse slot="content" class="tmx-collapse" v-model="value1">
        <Panel name="1">
          第二层
          <p slot="content">内容</p>
        </Panel>
        <Panel name="2">
          第二层
          <Collapse slot="content" class="tmx-collapse" v-model="value1">

            </Panel>
          </Collapse>
        </Panel>
      </Collapse>
    </Panel>
  </Collapse>
</template>
