    新增了两个属性 direction 和 column， column 只有在 direction="column"的时候才会生效

    如果需要设置silder的高度，直接设style即可

    <template>
  <div stype="margin-bottom: 30px;">
    <p style="padding: 20px 0; margin-left: 50px; display:flex;">
      <Slider :value="24" show-tip="always" :showLabel="true" direction="column" column="left"></Slider>
      <Slider :value="24" :showLabel="true" style="margin-left: 200px;height:100px" show-tip="always" direction="column"></Slider>
      <Slider :value="24" :showLabel="true" style="margin-left: 200px;height:200px" disabled  show-tip="always" direction="column"></Slider>
      <Slider :value="24" :showLabel="true" style="margin-left: 200px;height:300px"  disabled  show-tip="always" direction="column"></Slider>
    </p>
    <!-- <Slider range disabled></Slider> -->
  </div>
</template>
