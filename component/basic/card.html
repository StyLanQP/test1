  </TmxCard>
</template>

    # 一级卡片 -- 扩展 title + description + button

    Title 卡片 使用属性 class="tmx-card-action" type="First"
    卡头 使用属性 class="tmx-card-head" slot="title"
    扩展 title + description + button 的一级卡片
    <template>
  <div style="margin-top: 16px; background: #eee; padding:16px">
    <TmxCard class="tmx-card-action" type="First">
      <div slot="title" class="tmx-card-head">
        <span>Title</span>
        <span class="tmx-card-desc">
          卡片 使用属性 class="tmx-card-action" type="First"
          <br/>
          卡头 使用属性 class="tmx-card-head" slot="title" 
        </span>
        <Button class="tmx-btn" type="primary">Save</Button>
        <Button class="tmx-btn" style="margin-left:10px;" type="default">Cancel</Button>
      </div>
      <div style="height: 60px">扩展 title + description + button 的一级卡片</div>
    </TmxCard>
  </div>
</template>

    # 一级卡片 -- 扩展 title + button + hover

    Title
    卡片 使用属性 class="tmx-card-action" type="First"
    卡头 使用属性 slot="title"
    卡扩展 使用属性 slot="extra" 
    <template>
  <div style="margin-top: 16px">
    <TmxCard class="tmx-card-action" type="First" hover>
      <div slot="title">Title</div>
      <div slot="extra">
        <RadioGroup type="button">
          <Radio label="Status"></Radio>
          <Radio label="Model"></Radio>
        </RadioGroup>
      </div>
      <div style="height: 60px">
        卡片 使用属性 class="tmx-card-action" type="First"
        <br/>
        卡头 使用属性 slot="title" 
        <br/>
        卡扩展 使用属性 slot="extra" 
      </div>
    </TmxCard>
  </div>
</template>
