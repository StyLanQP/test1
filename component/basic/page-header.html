    # 基础用法

    Title 
    <template>
  <TmxPageHeader title="Title"> </TmxPageHeader>
</template>

    # 带操作按钮

    Title 
    <template>
  <TmxPageHeader title="Title">
    <button class="tmx-btn" type="primary">Save</button>
    <button class="tmx-btn">Cancel</button>
  </TmxPageHeader>
</template>
