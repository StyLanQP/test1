    # 基础用法，带背景色的提示

    <template>
  <div>
    <Button @click="background('flowInfo')">显示普通提示</Button>
    <Button @click="background('flowSuccess')">显示成功提示</Button>
    <Button @click="background('flowWarning')">显示警告提示</Button>
    <Button @click="background('flowError')">显示错误提示</Button>
  </div>
</template>

<script>
export default {
  methods: {
    background(type) {
        this.$Message[type]({
            background: true,
            duration: 5,
            content: "这是一条带背景色的通知",
      });
    },
  },
};
</script>

    # 跑马灯效果的提示

    <template>
  <div style="margin: 30px 0;">
    <Button @click="background1('info')">info</Button>
    <Button @click="background1('success')">success</Button>
    <Button @click="background1('warning')">waring</Button>
    <Button @click="background1('error')">error</Button>
  </div>
</template>

<script>
export default {
  methods: {
    background1(type) {
      const msg = this.$Message[type]({
        duration: 20,
        content: "Connecting",
        isSlide: true,
      });
    },
  },
};
</script>
