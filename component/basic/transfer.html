    代码示例 
    <template>
    <Transfer
      :data="data1"
      :target-keys="targetKeys1"
      :render-format="render1"
      @on-change="handleChange1"
      @on-selected-change="onSelectedChange"
    ></Transfer>
</template>

    <script>
  export default {
    data() {
      return {
        data1: this.getMockData(),
        targetKeys1: this.getTargetKeys(),
        tableHeight: 250,
        filterColumns: [
          {
            title: "Function Library",
            slot: "title",
          },
          // {
          //   title: 'Action',
          //   key: 'action',
          //   slot: 'action',
          //   width: 85,
          // },
        ],
      };
    },
    methods: {
      getMockData() {
        let mockData = [];
        for (let i = 1; i <= 20; i++) {
          mockData.push({
            key: i.toString(),
            label: "Content " + i,
            description: "The desc of content  " + i,
            disabled: Math.random() * 3 < 1,
          });
        }
        return mockData;
      },
      getTargetKeys() {
        return this.getMockData()
          .filter(() => Math.random() * 2 > 1)
          .map((item) => item.key);
      },
      render1(item) {
        return item.label;
      },
      handleChange1(newTargetKeys, direction, moveKeys) {
        console.log("handleChange1", newTargetKeys);
        console.log("handleChange2", direction);
        console.log("handleChange3", moveKeys);
        this.targetKeys1 = newTargetKeys;
      },
      onSelectedChange(sourceSelectedKeys, targetSelectedKeys) {
        console.log("sourceSelectedKeys", sourceSelectedKeys, targetSelectedKeys);
      },

    },
  };
</script>
