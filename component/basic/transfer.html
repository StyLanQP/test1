    <template>
    <div style="margin: 16px 0">
        <TmxTransfer 
            selectKey="title"
            rightTopTitle="selected"
            :data="dataset"
            :columns="filterColumns" 
            :initData="initData"
            :drag="true"
            :itemLink="true"
            :max-count="30"
            empty-text="no data"
            :tableHeight="tableHeight"
            @on-transfer-change="transferChangeHandler"
            @on-item-click="onClick"
            @on-max="onMax"
        >
            <template slot="filters">
                <TmxTableHeader>
                    <template slot="header-start">
                        <Input
                            v-model="listQuery.keyword"
                            class="input flex-item tmx-input"
                            prefix="ios-search"
                            clearable
                            placeholder="sim_card_search"
                            @on-search="filterChange"
                            @on-clear="filterChange"
                        />
                    </template>
                </TmxTableHeader>
            </template>
            <template slot="pagination">
              <Page class="tmx-page tmx-page-border" v-model="listQuery" :page-otions="{ sizer: false }" />
            </template>
        </TmxTransfer>
    </div>
</template>
