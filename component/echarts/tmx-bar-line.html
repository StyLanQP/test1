    # 使用

    # 1. 示例一（折线图）


    <template>
    <LineBar :chartData="lineBarData" style="border: 1px solid;padding: 40px 24px 0 10px;width: 100%; height: 260px"></LineBar>
</template>


    <template>
    <LineBar :chartData="lineBarData5" style="border: 1px solid;padding: 40px 24px 0 10px;width: 100%; height: 260px"></LineBar>
</template>


    export default {
  data() {
    return {
      lineBarData: {
        xData: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
        yData: [
          // 对应于echarts 的series, 可以自定义配置
          {
            name: "销量3",
            type: "line",
            smooth: true,
            itemStyle: { color: "#00CC9E" },
            data: [20, 32, 11, 34, 90, 30],
          },
        ],
      },
      lineBarData5: {
              xData: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
              yData: [
              {
                  name: "销量3",
                  type: "line",
                  // smooth: true,
                  // itemStyle: { color: "#00CC9E" },
                  areaStyle: {
                    color: {
                      type: 'linear',
                      x: 0,
                      y: 0,
                      x2: 0,
                      y2: 1,
                      colorStops: [
                        {
                          offset: 0,
                          color: 'rgba(51, 117, 255, 0.4)' // 0% 处的颜色
                        },
                        {
                          offset: 0.5,
                          color: 'rgba(51, 117, 255, 0.2)' // 0% 处的颜色
                        },
                        {
                          offset: 1,
                          color: 'rgba(51, 117, 255, 0)' // 100% 处的颜色
                        }
                      ],
                      global: false // 缺省为 false
                    }
                  },
                  data: [20, 32, 11, 34, 90, 30],
              },
              ],
          }
    };
  },
};

    # 2. 示例三(自定义样式)


    自定义样式， 比如每个数据的颜色、 自定义 options 样式、 把数据转换单位等

    <template>
    <LineBar :chartData="lineBarData" style="border: 1px solid;padding: 40px 24px 0 10px;width: 100%; height: 260px"></LineBar>
</template>


    export default {
  data() {
    return {
      lineBarData: {
        options: {
          grid: {
            //自定义边距
            left: 60,
          },
          yAxis: {
            //y轴显示方式
            axisLabel: {
              show: true,
              color: "#8F9BB2",
              formatter: (value) => {
                // 数值转为流量单位
                return bytesToSize(parseInt(value));
              },
            },
          },
        },
        toUnitFn: bytesToSize, // 把tooltip数值转换单位(Function)
        yData: [
          {
            name: "销量1",
            type: "bar",
            data: [
              { value: 121, name: "羊毛衫", itemStyle: { color: "red" } }, // 自定义个别柱子颜色
              { value: 21, name: "衬衫" },
              { value: 21, name: "雪纺衫" },
              { value: 21, name: "裤子" },
              { value: 21, name: "高跟鞋" },
              { value: 21, name: "袜子" },
            ],
          },
          { name: "销量2", type: "bar", data: [15, 10, 16, 30, 40, 40] },
          { name: "销量3", type: "line", data: [20, 32, 11, 34, 90, 30] },
        ],
      },
    };
  },
};
