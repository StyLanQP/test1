    # 2. 示例二 (自定义样式)

    
<WarningLine style="border: 1px solid;padding: 40px 24px 0 10px;width: 100%; height: 260px" :chartData="barData"></WarningLine>


    const bytesToSize = (bytes) => {
  if (bytes === 0) return "0 B";
  if (!bytes) return "--";
  var k = 600; // or 1024
  var sizes = ["B", "KB", "MB", "GB", "TB", "PB", "EB", "ZB", "YB"];
  var i = Math.floor(Math.log(bytes) / Math.log(k));

  return (bytes / Math.pow(k, i)).toFixed(2) + sizes[i];
};

export default {
  data() {
    return {
      barData: {
        warningLine: 600, //警戒值
        markPoint: {
          // 警戒线的样式（具体看api，自定义会合并）
          data: [
            {
              yAxis: 600,
              x: "100%",
              symbolOffset: [-100, -15],
              label: {
                textStyle: { color: "#FF6666" },
                fontSize: 12,
                position: "right",
                formatter: `Mean ${600} KB`,
              },
            },
          ],
        },
        options: {
          grid: {
            top: 20,
            left: "5%",
            right: "2%",
          },
          yAxis: {
            type: "value",
            axisLabel: { show: false, color: "#8F9BB2" },
          },
        },
        toUnitFn: bytesToSize, // 单位转换方法
        // xData: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
        yData: [
          {
            type: "line",
            name: "hhh",
            data: [
              { value: 548, name: "搜索引擎2" },
              { value: 148, name: "搜索引擎w" },
              { value: 448, name: "搜索引擎4" },
              { value: 908, name: "搜索引擎5" },
              { value: 848, name: "搜索引擎3" },
              { value: 748, name: "搜索引擎1" },
              { value: 648, name: "搜索引擎" },
              { value: 548, name: "搜索引擎e" },
            ],
          },
        ],
      },
    };
  },
};
