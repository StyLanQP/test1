    # 使用

    这个组件的定制化比较高，能修改空间不大

    <RankBar :chartData="rankBar" style="border: 1px solid;padding: 40px 24px 0 10px;width: 100%; height: 530px;"></RankBar>

    }
}

    # 2. 示例二

    <template>
    <RankBar :chartData="rankBar1" :rankNum="7" style="border: 1px solid;padding: 40px 24px 0 10px;width: 100%; height: 530px;"></RankBar>
</template>


    const bytesToSize = bytes => {
  if (bytes === 0) return '0 B'
  if (!bytes) return '--'
  var k = 1000 // or 1024
  var sizes = ['B', 'KB', 'MB', 'GB', 'TB', 'PB', 'EB', 'ZB', 'YB']
  var i = Math.floor(Math.log(bytes) / Math.log(k))

  return (bytes / Math.pow(k, i)).toFixed(2) + sizes[i]
}
export default {
    data() {
        return {
            rankBar1: {
                toUnitFn: bytesToSize,
                yData: [
                    {
                        name: "hhhh",
                        data: [
                            { value: 12875536,name: "衬衫1"},
                            {value: 1214,name: "衬衫2"},
                            {value: 1129,name: "衬衫3"},
                            {value: 925,name: "衬衫4"},
                        ],
                    }
                ],
                colorList: ["#1555D9","#3375FF","#4D86FF","#6697FF","#80A9FF","#80A9FF","#B3CBFF","#CCDDFF","#CCDDFF","#F2F6FF"],
            },
        }
    }
}
