    # 1. 示例一

    • dd
    • ee
    <template>
 <TreeMap style="border: 1px solid;padding: 40px 24px 0 10px;width: 392px; height: 260px":chartData="chartData"></TreeMap>
</template>


    export default {
  data() {
    return {
      chartData: {
        yData: [
          { name: "aa", value: 33 },
          { name: "bb", value: 33 },
          { name: "cc", value: 33 },
          { name: "dd", value: 33 },
          { name: "ee", value: 33 },

        colorList: [

            "#FF73BC",
            "#FFAC00",
            "#A380FF",
            "#FF6666",
            "#00CC9E",
        ]
      },
    };
  },
};
