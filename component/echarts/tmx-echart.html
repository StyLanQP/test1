    # 一、版本要求

    和tmx项目保持一致使用 echarts4（建议统一安装echarts@4.9.0）

    # 二、样式说明


    所有封装组件统一自适应 “width：100%; height: 1000% ”，请注意设置父容器宽高

    # 三、 安装使用

    # 1. 使用默认的实例化组件（如果图表实例化与项目业务无关，建议使用默认方式）


    组件库自带默认的初始化组件 ：EchartsComp， 并且自带UI统一主题，也会抛出初始化后的echarts实例

    # 初始化方式：

    import TmxUI from 'tmx-ui'
import echarts from 'echarts'

// 在tmx-ui引用echarts
TmxUI.TmxEcharts.init(Vue, echarts)


    <template>
  <LineBar :chartData="lineBarData" @onInitEchart="onInitEchart"></LineBar>
</template>

    接收echarts 实例的回调函数是onInitEchart， 可以监听各种事件，具体请看echarts 官网

    onInitEchart(val) {
    this.myEchart = val
      const that = this
      //   监听柱状图点击事件
      this.myEchart &&
        this.myEchart.on('click', function(params) {
        //   that.goToDetail({ iccid: params.name })
        })

    // 监听整个图表点击事件
    this.myEchart &&
    this.myEchart.getZr().on('click', this.echartClick)
}

    # 2. 使用项目自定义的初始化组件

    [1] 如果项目需要用到自定义的 EchartsComp，不需要执行上面操作， 但是需要在使用组件时传入自定义的EchartsCom

    <template>
   <LineBar :chartData="lineBarData">
       // 此处写法和方式固定
       <template v-slot:echartsCom="param">
          // 自定义的组件
           <EchartsComp :options="param.options"></EchartsComp>
       </template>
   </LineBar>
</template>

    [2] 初始化统一的主题样式

    import {ThemeJSON} from 'tmx-ui'

// 在echarts init的时候传入主题
this.$echarts.registerTheme("customed", themeJSON.theme);
this.chartDom = this.$echarts.init(ele, "customed");

    # 3. 传参特别声明

    除了定制化特别高的组件，其他组件都接收 options（会合并） 来自定义业务样式

    # 例如正常柱状图传参如下
    lineBarData= {
    // xData: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
    yData: [
        { name: "销量2", type: "bar", data: [15, 10, 16, 30, 40, 40] },
        { name: "销量3", type: "line", data: [20, 32, 11, 34, 90, 30] },
    ],
}
    # 定制化x坐标轴样式（和echarts官网Api保持一致）
    lineBarData= {
    // 自定义业务的options样式
    options: {
        xAxis: { // 定制x坐标轴样式
            axisLabel: {
                // 显示所有标签
                interval: 0,
                formatter: (value) => {
                const arr = value.split('-').slice(-2)
                return arr.join('-')
                }
            }
        },
    },
    yData: [
        { name: "销量2", type: "bar", data: [15, 10, 16, 30, 40, 40] },
        { name: "销量3", type: "line", data: [20, 32, 11, 34, 90, 30] },
    ],
}
    # 定制化tooltip（和echarts官网Api保持一致）
    // 如果需要传参为下面的格式
lineBarData= {
    // 自定义业务的options样式
    options: {
        tooltip: {
            show: false
        }
    },
    yData: [
        { name: "销量2", type: "bar", data: [15, 10, 16, 30, 40, 40] },
        { name: "销量3", type: "line", data: [20, 32, 11, 34, 90, 30] },
    ],
}
